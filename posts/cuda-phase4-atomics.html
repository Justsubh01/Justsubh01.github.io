CUDA Mastery Phase 4.4 — Atomic Operations: Thread-Safe Updates Without Locks

How atomic operations let thousands of threads safely update shared values, which atomics are available, why naive use serializes your entire kernel, and the histogram example that shows the right way to handle contention.

When thousands of GPU threads try to update the same memory location — a counter, a histogram bin, a global maximum — you have a race condition. On the CPU you'd use a mutex. On the GPU, mutexes would serialize all 32,000 threads into a queue. CUDA's atomic operations solve this without locks, but they come with a performance cost you need to understand and mitigate.

1. The Race Condition Problem

  // WRONG: race condition — thousands of threads updating the same counter
  __global__ void count_nonzero(float *data, int *count, int n) {
      int idx = blockIdx.x * blockDim.x + threadIdx.x;
      if (idx < n && data[idx] != 0.0f) {
          (*count)++;    // READ-MODIFY-WRITE: not atomic! Race condition!
      }
  }

  // What happens at hardware level:
  // Thread A:  load *count → 5
  // Thread B:  load *count → 5   (before A writes back!)
  // Thread A:  store 5+1=6 → *count
  // Thread B:  store 5+1=6 → *count  (overwrites A's result!)
  // Final: *count = 6 instead of 7 — one increment lost!

2. atomicAdd — The Most Common Atomic

  // CORRECT: atomic increment
  __global__ void count_nonzero(float *data, int *count, int n) {
      int idx = blockIdx.x * blockDim.x + threadIdx.x;
      if (idx < n && data[idx] != 0.0f) {
          atomicAdd(count, 1);   // Atomic: read-modify-write as one indivisible operation
      }
  }

  // atomicAdd returns the OLD value (before the add):
  int old = atomicAdd(count, 1);
  // old = value before this thread's increment
  // *count = old + 1 after this thread's operation

3. All Available Atomics

  // Integer atomics (int, unsigned int, unsigned long long):
  int old;
  old = atomicAdd(addr, val);          // *addr += val
  old = atomicSub(addr, val);          // *addr -= val
  old = atomicExch(addr, val);         // *addr = val (exchange/swap)
  old = atomicMin(addr, val);          // *addr = min(*addr, val)
  old = atomicMax(addr, val);          // *addr = max(*addr, val)
  old = atomicInc(addr, maxval);       // *addr = (*addr>=maxval) ? 0 : (*addr+1)
  old = atomicDec(addr, maxval);       // *addr = (*addr==0||*addr>maxval) ? maxval : (*addr-1)
  old = atomicAnd(addr, val);          // *addr &= val
  old = atomicOr (addr, val);          // *addr |= val
  old = atomicXor(addr, val);          // *addr ^= val
  old = atomicCAS(addr, compare, val); // Compare-And-Swap (most powerful!)

  // Float atomics (compute capability 2.0+):
  old = atomicAdd((float*)addr, val);  // float add — hardware atomic since sm_20

  // Double atomics (compute capability 6.0+):
  old = atomicAdd((double*)addr, val); // double add

4. atomicCAS — The Universal Atomic

  // atomicCAS: Compare-And-Swap
  // If *addr == compare: set *addr = val, return old value
  // If *addr != compare: do nothing, return old value
  // This is the foundation for implementing ANY custom atomic operation.

  // Example: implement atomic float max (not natively available as atomic):
  __device__ float atomicMaxFloat(float *addr, float val) {
      float old = *addr;
      while (val > old) {
          float assumed = old;
          old = __int_as_float(
              atomicCAS((int*)addr,
                        __float_as_int(assumed),
                        __float_as_int(val)));
          if (old == assumed) break;  // CAS succeeded
          // If old != assumed: another thread changed it; retry with new old
      }
      return old;
  }

  // Example: atomic lock (spinlock) — don't use in production, but illustrative:
  __device__ void lock(int *mutex) {
      while (atomicCAS(mutex, 0, 1) != 0);  // spin until we set 0→1
  }
  __device__ void unlock(int *mutex) {
      atomicExch(mutex, 0);   // set back to 0
  }

5. The Histogram — Atomics in Practice

  // Version 1: Naive — heavy contention!
  __global__ void histogram_naive(unsigned char *data, int *hist, int n) {
      int idx = blockIdx.x * blockDim.x + threadIdx.x;
      if (idx < n) {
          atomicAdd(&hist[data[idx]], 1);  // 256 possible bins
          // If data has many pixels with value 128, ALL threads hammer hist[128]
          // → Sequential: one thread at a time → terrible performance!
      }
  }
  // Version 2: Shared memory histogram — reduce contention!
  __global__ void histogram_shared(unsigned char *data, int *hist, int n) {
      __shared__ int s_hist[256];

      // Initialize shared histogram:
      if (threadIdx.x < 256) s_hist[threadIdx.x] = 0;
      __syncthreads();

      // Each thread updates SHARED histogram (fast — on-chip):
      int idx = blockIdx.x * blockDim.x + threadIdx.x;
      if (idx < n) {
          atomicAdd(&s_hist[data[idx]], 1);  // shared memory atomic — fast!
      }
      __syncthreads();

      // One write per bin per block to global memory (256 atomics total):
      if (threadIdx.x < 256) {
          atomicAdd(&hist[threadIdx.x], s_hist[threadIdx.x]);
      }
  }

  // Why this is faster:
  // Naive:  n atomic operations on global memory (millions)
  // Shared: n atomics on shared memory (fast) + 256 × num_blocks on global
  // With n=1M, blockSize=256, gridSize=4096: 4096×256=1M global → 256×4096=1M shared
  // But shared atomics are ~10× faster!
  // Net: ~10× speedup over naive

6. Private Copies Pattern

  // Pattern: each thread accumulates its own private copy,
  // then atomically merges into the global result.
  // Zero contention during computation, one atomic at the end.

  __global__ void sum_reduce_private(float *data, float *result, int n) {
      int idx = blockIdx.x * blockDim.x + threadIdx.x;
      int stride = blockDim.x * gridDim.x;

      // Each thread accumulates privately:
      float private_sum = 0.0f;
      for (int i = idx; i < n; i += stride)
          private_sum += data[i];

      // One atomic per thread to merge (n/gridSize atomics total instead of n):
      atomicAdd(result, private_sum);
  }

  // But wait — this still has contention at the atomicAdd.
  // Better: use warp-level reduction first, then one atomic per warp:
  __global__ void sum_reduce_warp(float *data, float *result, int n) {
      int idx = blockIdx.x * blockDim.x + threadIdx.x;
      int stride = blockDim.x * gridDim.x;

      float val = 0.0f;
      for (int i = idx; i < n; i += stride) val += data[i];

      // Warp reduction (no atomics within warp):
      for (int offset = 16; offset > 0; offset /= 2)
          val += __shfl_down_sync(0xffffffff, val, offset);

      // Only thread 0 of each warp does the atomic:
      if (threadIdx.x % 32 == 0)
          atomicAdd(result, val);
      // Atomics: (n/blockDim.x/32) instead of n → 1024× less contention!
  }

7. Performance: When Atomics Hurt

  Atomics are serialized at the hardware level.
  If K threads atomicAdd to the same address simultaneously:
    → K sequential operations instead of 1 parallel one
    → K× slower than a non-atomic add

  Contention levels:
  ─────────────────────────────────────────────────────────
  Low contention  (few threads per address): ~1–2 cycles
  High contention (many threads per address): up to 100+ cycles
  Worst case: all 1024 threads in a block atomicAdd to same addr → 1024× slowdown

  Rules:
  ✓ Use private accumulation + final atomic merge
  ✓ Use shared memory for partial histograms (local contention only)
  ✓ Use warp reduction before atomicAdd (reduces atomics by 32×)
  ✗ Don't atomicAdd in an inner loop if the address is frequently hit
  ✗ Don't assume float atomics are as fast as int atomics (they're not on older GPUs)

Takeaways

  • Race conditions occur when multiple threads read-modify-write the same address. Atomics fix this.
  • atomicAdd() is the most common; it returns the old value before the addition.
  • atomicCAS() is the universal building block for implementing any custom atomic.
  • High contention on one address serializes all threads hitting it — can be 1000× slower.
  • Shared memory histogram: accumulate locally in shared memory, merge to global at end.
  • Warp reduction before atomic: reduce 32 values per warp first, then one atomic per warp.
  • Private copies pattern: each thread accumulates privately, then one final atomic merge.
