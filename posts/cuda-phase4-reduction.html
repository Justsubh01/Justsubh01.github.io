CUDA Mastery Phase 4.2 — Parallel Reduction: The Classic Interview Problem, Solved 6 Ways

The parallel sum reduction — from naive sequential loop to fully optimized warp-unrolled GPU kernel — is the most commonly asked CUDA interview question. Here are six progressively better implementations with the exact reasoning behind every optimization.

If you only learn one parallel algorithm in CUDA, make it reduction. It appears everywhere: sum, max, min, dot product, mean, variance. And the journey from the naive implementation to the optimized one teaches you warp divergence, memory coalescing, shared memory, and occupancy all at once. NVIDIA has a famous paper "Optimizing Parallel Reduction in CUDA" — this post is my walkthrough of those ideas.

The Problem

  Sum N = 1,048,576 (1M) floats in parallel.

  CPU baseline: O(N) sequential additions.
  for (int i = 1; i < N; i++) total += data[i];
  → Fast enough for small N, but 1M floats on CPU: ~1ms.

  GPU target: use massive parallelism to reduce to O(log N) steps.
  Each step halves the problem: [N → N/2 → N/4 → ... → 1]
  log2(1M) = 20 steps with 1M/2 parallel additions each step.
  Potential speedup: 1M / 20 = 50,000× (vs CPU, ignoring latency)

Reduction 1 — Interleaved Addressing (Naive)

__global__ void reduce1(float *g_idata, float *g_odata, int n) {
    extern __shared__ float sdata[];
    int tid = threadIdx.x;
    int i   = blockIdx.x * blockDim.x + threadIdx.x;

    sdata[tid] = (i < n) ? g_idata[i] : 0.0f;
    __syncthreads();

    // Reduction in shared memory — interleaved addressing:
    for (int s = 1; s < blockDim.x; s *= 2) {
        if (tid % (2 * s) == 0) {            // ← PROBLEM: modulo is slow + divergence!
            sdata[tid] += sdata[tid + s];
        }
        __syncthreads();
    }

    if (tid == 0) g_odata[blockIdx.x] = sdata[0];
}

// Problems:
// 1. tid % (2*s) == 0: only half threads active, rest are IDLE (bad occupancy)
// 2. Highly divergent: condition varies within a warp
// 3. % operator is expensive (integer divide)
  Visualization with 8 elements [8 6 4 2 7 3 5 1]:
  Step s=1 (active threads: 0,2,4,6):
    sdata[0] += sdata[1]  →  14
    sdata[2] += sdata[3]  →   6
    sdata[4] += sdata[5]  →  10
    sdata[6] += sdata[7]  →   6
    Result: [14 6 6 2 10 3 6 1]

  Step s=2 (active threads: 0,4):
    sdata[0] += sdata[2]  →  20
    sdata[4] += sdata[6]  →  16
    Result: [20 6 6 2 16 3 6 1]

  Step s=4 (active thread: 0):
    sdata[0] += sdata[4]  →  36
    Sum = 36 = 8+6+4+2+7+3+5+1 ✓

Reduction 2 — Non-Divergent Interleaved (Fix Divergence)

__global__ void reduce2(float *g_idata, float *g_odata, int n) {
    extern __shared__ float sdata[];
    int tid = threadIdx.x;
    int i   = blockIdx.x * blockDim.x + threadIdx.x;

    sdata[tid] = (i < n) ? g_idata[i] : 0.0f;
    __syncthreads();

    // Fix: use stride-based indexing instead of modulo
    for (int s = 1; s < blockDim.x; s *= 2) {
        int index = 2 * s * tid;          // ← no modulo, no divergence
        if (index + s < blockDim.x) {
            sdata[index] += sdata[index + s];
        }
        __syncthreads();
    }

    if (tid == 0) g_odata[blockIdx.x] = sdata[0];
}

// Better: no divergence within warps for most steps.
// Remaining problem: shared memory bank conflicts (scattered access pattern).

Reduction 3 — Sequential Addressing (Fix Bank Conflicts)

__global__ void reduce3(float *g_idata, float *g_odata, int n) {
    extern __shared__ float sdata[];
    int tid = threadIdx.x;
    int i   = blockIdx.x * blockDim.x + threadIdx.x;

    sdata[tid] = (i < n) ? g_idata[i] : 0.0f;
    __syncthreads();

    // Sequential addressing: stride starts at blockDim/2, halves each step
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            sdata[tid] += sdata[tid + s];   // ← contiguous, no bank conflicts!
        }
        __syncthreads();
    }

    if (tid == 0) g_odata[blockIdx.x] = sdata[0];
}

// Why this is better:
// - sdata[tid] and sdata[tid+s] have no bank conflicts (consecutive banks)
// - Half the threads are idle per step (still can be improved)
  Sequential addressing with 8 elements [8 6 4 2 7 3 5 1]:
  Step s=4 (threads 0,1,2,3 active):
    tid=0: sdata[0] += sdata[4]  →  8+7=15
    tid=1: sdata[1] += sdata[5]  →  6+3=9
    tid=2: sdata[2] += sdata[6]  →  4+5=9
    tid=3: sdata[3] += sdata[7]  →  2+1=3
    Result: [15 9 9 3 7 3 5 1]

  Step s=2 (threads 0,1 active):
    sdata[0] += sdata[2]  →  15+9=24
    sdata[1] += sdata[3]  →  9+3=12
    Result: [24 12 9 3 ...]

  Step s=1 (thread 0 active):
    sdata[0] += sdata[1]  →  24+12=36 ✓

Reduction 4 — First Add During Load (Halve the Blocks)

__global__ void reduce4(float *g_idata, float *g_odata, int n) {
    extern __shared__ float sdata[];
    int tid = threadIdx.x;
    int i   = blockIdx.x * (blockDim.x * 2) + threadIdx.x;  // ← NOTE: *2!

    // Each thread loads TWO elements and adds them immediately:
    float v = 0.0f;
    if (i < n)       v  = g_idata[i];
    if (i + blockDim.x < n) v += g_idata[i + blockDim.x];   // ← first add during load!
    sdata[tid] = v;
    __syncthreads();

    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) sdata[tid] += sdata[tid + s];
        __syncthreads();
    }

    if (tid == 0) g_odata[blockIdx.x] = sdata[0];
}

// Why this is better:
// With blockDim=256, we now process 512 elements per block instead of 256.
// Need half as many blocks → half the launch overhead.
// Each thread does 1 extra ADD instead of 1 idle thread slot.
// Net effect: ~2× faster than reduce3.

Reduction 5 — Unroll the Last Warp

// When s <= 32, we're working within a SINGLE WARP.
// Threads in a warp execute in lockstep — no __syncthreads() needed!
// We can unroll these steps manually to save synchronization overhead.

__device__ void warpReduce(volatile float* sdata, int tid) {
    // 'volatile' prevents compiler from optimizing away these writes
    sdata[tid] += sdata[tid + 32];
    sdata[tid] += sdata[tid + 16];
    sdata[tid] += sdata[tid +  8];
    sdata[tid] += sdata[tid +  4];
    sdata[tid] += sdata[tid +  2];
    sdata[tid] += sdata[tid +  1];
}

__global__ void reduce5(float *g_idata, float *g_odata, int n) {
    extern __shared__ float sdata[];
    int tid = threadIdx.x;
    int i   = blockIdx.x * (blockDim.x * 2) + threadIdx.x;

    float v = 0.0f;
    if (i < n) v = g_idata[i];
    if (i + blockDim.x < n) v += g_idata[i + blockDim.x];
    sdata[tid] = v;
    __syncthreads();

    // Unroll all steps above warp size:
    for (int s = blockDim.x / 2; s > 32; s >>= 1) {
        if (tid < s) sdata[tid] += sdata[tid + s];
        __syncthreads();
    }

    // Last 6 steps: within a single warp, no __syncthreads():
    if (tid < 32) warpReduce(sdata, tid);

    if (tid == 0) g_odata[blockIdx.x] = sdata[0];
}

// Saves 5 __syncthreads() calls (expensive: stall all threads until slowest)
// Notable: volatile ensures stores are visible to other threads in warp

Reduction 6 — Warp Shuffle (No Shared Memory for Last Warp)

__device__ float warpReduceShfl(float val) {
    for (int offset = 16; offset > 0; offset /= 2)
        val += __shfl_down_sync(0xffffffff, val, offset);
    return val;  // thread 0 has the sum
}

__global__ void reduce6(float *g_idata, float *g_odata, int n) {
    extern __shared__ float sdata[];
    int tid = threadIdx.x;
    int i   = blockIdx.x * (blockDim.x * 2) + threadIdx.x;

    float val = 0.0f;
    if (i < n) val = g_idata[i];
    if (i + blockDim.x < n) val += g_idata[i + blockDim.x];

    // Warp-level reduction using shuffle (no shared memory for this part):
    val = warpReduceShfl(val);

    // One thread per warp writes to shared memory:
    int laneId = tid % 32;
    int warpId = tid / 32;
    if (laneId == 0) sdata[warpId] = val;
    __syncthreads();

    // Final reduction across warps (only first warp needed):
    val = (tid < blockDim.x / 32) ? sdata[tid] : 0.0f;
    if (warpId == 0) val = warpReduceShfl(val);
    if (tid == 0) g_odata[blockIdx.x] = val;
}

// Uses only 1 shared memory read/write per warp (vs per thread in earlier versions)
// __shfl_down_sync is faster than shared memory for warp-level reductions

Performance Summary

  Reduction version comparison (N=8M, blockDim=256, RTX 3090):
  ─────────────────────────────────────────────────────────────────
  Version  | Key optimization          | Time  | Bandwidth
  ─────────┼───────────────────────────┼───────┼──────────────────

  2 Fix    | no modulo, less divergence| 5.1ms | 52 GB/s

  4 Load   | first add during load     | 1.6ms | 165 GB/s

Using CUB for Production Code

#include <cub/cub.cuh>

// CUB device-wide reduction:
float *d_data;
float *d_result;
int    n = 1 << 20;

// Get temp storage size:
void   *d_temp = nullptr;
size_t  temp_bytes = 0;
cub::DeviceReduce::Sum(d_temp, temp_bytes, d_data, d_result, n);

// Allocate temp storage:
cudaMalloc(&d_temp, temp_bytes);

// Run reduction:
cub::DeviceReduce::Sum(d_temp, temp_bytes, d_data, d_result, n);
cudaDeviceSynchronize();

// d_result[0] now holds the sum.

// CUB also provides: Reduce (custom op), ArgMin, ArgMax, SegmentedReduce
// Use CUB in production — it's always at or near hardware peak bandwidth.

Takeaways

  • Parallel reduction: O(N) work, O(log N) steps — matches sequential work complexity.
  • Version 1→3: fix divergence and bank conflicts via sequential addressing.
  • Version 4: add two elements during load → half as many blocks needed.
  • Version 5: unroll the last warp to eliminate unnecessary __syncthreads() calls.
  • Version 6: use __shfl_down_sync() for the last warp — avoids shared memory entirely.
  • 7× speedup from version 1 to version 6 on the same hardware.
  • In production: use CUB (cub::DeviceReduce::Sum) — library-optimized, always near peak.
