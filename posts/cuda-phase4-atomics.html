Atomic Operations — Thread-Safe Updates Without Locks

Atomics solve the race condition problem: when thousands of threads want to update the same memory location simultaneously, you need a guarantee that each update is indivisible. CUDA provides hardware-accelerated atomics for integers, floats, and custom compare-and-swap operations.

Race condition: two threads read a counter (value=5), both add 1, both write back 6. Result: 6, not 7. With 10,000 threads all incrementing the same counter, you might get anything from 1 to 10,000. Atomics fix this.

What Are Atomics and Why They're Needed

An atomic operation reads, modifies, and writes a memory location as an indivisible unit. No other thread can see the intermediate state. The hardware guarantees that if 10,000 threads all atomically increment the same counter, the final result is exactly 10,000. 

// WRONG: race condition, result unpredictable
__global__ void wrongCounter(int* counter, int n) {
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i < n) {
        int old = *counter;   // read
        *counter = old + 1;   // write (another thread may write between read and write!)
    }
}

// CORRECT: atomic add guarantees each thread's increment is applied exactly once
__global__ void correctCounter(int* counter, int n) {
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i < n) {
        atomicAdd(counter, 1);  // read-modify-write as one hardware instruction
    }
}

__global__ void atomicExamples(int* data, unsigned* flags, float* fdata) {
    int i = blockIdx.x * blockDim.x + threadIdx.x;

    // Atomic add (returns old value)
    int old = atomicAdd(&data[0], 1);

    // Atomic max
    atomicMax(&data[1], i);      // data[1] = max of all thread indices

    // Atomic OR: set bit i in flags[0]
    atomicOr(&flags[0], (1u << (i % 32)));

    // Atomic exchange: replace value, get old
    int prev = atomicExch(&data[2], i);  // last writer wins

    // Float atomic add (sm_20+)
    atomicAdd(&fdata[0], 1.0f);
}

Compare-And-Swap — The Foundation

atomicCAS(addr, compare, val) is the primitive from which all other atomics can be built. It atomically does: if (*addr == compare) { *addr = val; } return old_value; 

// Build custom float atomicMin using CAS:
__device__ float atomicMinFloat(float* addr, float val) {
    int* addr_as_int = (int*)addr;
    int old = *addr_as_int, assumed;
    do {
        assumed = old;
        float old_float = __int_as_float(assumed);
        float new_float = fminf(old_float, val);
        int new_int = __float_as_int(new_float);
        old = atomicCAS(addr_as_int, assumed, new_int);
        // Loop until CAS succeeds (old == assumed means no one changed it between read and CAS)
    } while (assumed != old);
    return __int_as_float(old);
}

// CAS-based spinlock (DON'T use in production - can deadlock):
__device__ void lock(int* mutex) {
    while (atomicCAS(mutex, 0, 1) != 0);  // spin until we get 0->1
}
__device__ void unlock(int* mutex) {
    atomicExch(mutex, 0);  // release
}
Spinlocks in GPU code are dangerous. If a thread holding a lock is context-switched out (e.g., due to a page fault), all other threads spin indefinitely. Spinlocks also kill warp efficiency. Always prefer warp-aggregated atomics or reduction patterns over lock-based approaches.

Floating-Point Atomics

atomicAdd(float*, float) is hardware-native since sm_20. atomicAdd(double*, double) since sm_60. But there is no hardware atomicMin/Max for floats — use the CAS trick above. 

// Half-precision atomics (sm_70+ with appropriate flags):
// #include <cuda_fp16.h>
// atomicAdd(__half* addr, __half val)  -- native on Volta+

// Double precision atomic add:
__global__ void doublePrecisionReduction(double* data, double* result, int n) {
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i < n) {
        atomicAdd(result, data[i]);  // double atomicAdd, sm_60+
    }
}

Atomic Performance and Contention

The performance of atomics depends critically on contention — how many threads compete for the same address simultaneously.

# atomicAdd performance (A100, N=16M threads): Contention level | Throughput (atomic ops/sec) | vs no atomic 1 unique address | 0.6 billion ops/s | 0.03x (32x slower than compute) 32 unique addresses | 2.1 billion ops/s | 0.11x 1024 unique addresses | 18.7 billion ops/s | 1.0x (baseline) No atomic (pure add) | 18.4 billion ops/s | 1.0x

When all 2048 threads/SM hit the same address, the memory controller serializes them into 2048 sequential operations — killing throughput. When threads hit different addresses, atomics are essentially free.

The atomic golden rule: Maximize the number of unique addresses. For a histogram with 256 bins and 1M threads: 1M/256 = 3906 threads per bin on average. That's manageable. For a global counter updated by all threads: 1M threads, 1 address — use warp aggregation instead.

Warp Aggregation — Fast Atomics

Instead of each thread doing its own atomic, have the warp collaborate: count how many threads want to add, then only ONE thread does the atomic with the total. This reduces atomic contention by up to 32×. 

// Warp-aggregated atomic add (32x fewer atomic operations)
__device__ int atomicAggInc(int* counter) {
    unsigned mask = __activemask();           // which threads are active
    int leader = __ffs(mask) - 1;            // lowest active lane
    int laneId = threadIdx.x % 32;

    int res = 0;
    if (laneId == leader) {
        // Only one thread does the atomic, adds count for entire warp
        res = atomicAdd(counter, __popc(mask));  // __popc = popcount
    }

    // Broadcast leader's result to all active threads
    res = __shfl_sync(mask, res, leader);

    // Each thread's "slot" = base + its position among active threads
    return res + __popc(mask & ((1 << laneId) - 1));
}

// Usage: exactly equivalent to each thread calling atomicAdd(counter, 1)
// but 32x fewer actual atomic operations
__global__ void countActive(int* data, int* counter, int n) {
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i >= n) return;
    if (data[i] > 0) {
        int slot = atomicAggInc(counter);  // each active thread gets unique slot
        // use slot...
    }
}

Case Study: Parallel Histogram

// Naive: global atomics, high contention for popular bins
__global__ void histogramNaive(const int* data, int* hist, int n, int nBins) {
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i < n) {
        atomicAdd(&hist[data[i] % nBins], 1);  // all threads compete for same bins
    }
}

// Optimized: shared memory histogram per block, then atomicAdd to global
__global__ void histogramShared(const int* data, int* hist, int n, int nBins) {
    extern __shared__ int localHist[];  // launch with nBins*sizeof(int) smem

    // Initialize local histogram
    for (int b = threadIdx.x; b < nBins; b += blockDim.x)
        localHist[b] = 0;
    __syncthreads();

    // Accumulate into local histogram (fast shared memory atomics)
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i < n) {
        atomicAdd(&localHist[data[i] % nBins], 1);  // shared mem atomics = fast
    }
    __syncthreads();

    // Merge local into global (only gridDim.x * nBins total global atomics vs N)
    for (int b = threadIdx.x; b < nBins; b += blockDim.x)
        if (localHist[b] > 0)
            atomicAdd(&hist[b], localHist[b]);
}

// Performance (N=16M, nBins=256, A100):
// Naive global atomics:          8.2 ms
// Shared memory histogram:       0.9 ms  (9x faster)

Lock-Free Patterns

// Lock-free stack using CAS
struct GPUStack {
    int* data;
    int  top;  // atomic
};

__device__ void push(GPUStack* stack, int val) {
    int idx = atomicAdd(&stack->top, 1);  // claim slot
    stack->data[idx] = val;               // write (might be visible before atomicAdd elsewhere)
    // Note: this is actually not a correct lock-free stack without more care
    // Use thrust::device_vector for production
}

__device__ int pop(GPUStack* stack) {
    int idx = atomicSub(&stack->top, 1) - 1;  // claim slot
    if (idx < 0) { atomicAdd(&stack->top, 1); return -1; }  // empty
    return stack->data[idx];
}
