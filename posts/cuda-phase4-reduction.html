Parallel Reduction — Six Implementations from Naive to Near-Optimal

Reduction (computing a sum, max, or dot product across an array) is the most studied CUDA pattern. NVIDIA's Mark Harris wrote a famous paper showing 6 progressive optimizations that together achieve a 30× speedup. Understanding each step teaches every major CUDA optimization principle.

Reduction is deceptively simple: given N numbers, compute their sum. On a CPU: one loop, N-1 additions, O(N) time. On a GPU: we can use a tree to do it in O(log N) parallel steps. But getting to peak memory bandwidth efficiency requires 6 distinct optimizations, each teaching something important.

Tree reduction in 3 steps for 8 elements. Step 1: 4 pairs summed simultaneously. Step 2: 2 pairs. Step 3: final sum. Total: log₂(8) = 3 steps instead of 7 sequential additions. The speedup grows with N.

The Problem: Sum N Floats on the GPU

We want: result = data[0] + data[1] + ... + data[N-1]. The GPU parallelizes by having each block reduce a chunk of data to a single value, then summing the per-block results on CPU (or in a second kernel pass).

The fundamental GPU constraint: threads can only communicate directly within a block (via shared memory + __syncthreads). Cross-block communication requires global memory. So we do a two-level reduction: block-level in shared memory, then combine block results.

Reduction 1 — Interleaved Addressing (Naive, Divergent)

__global__ void reduce1(float* g_data, float* g_out, int n) {
    extern __shared__ float sdata[];
    int tid = threadIdx.x;
    int i   = blockIdx.x * blockDim.x + threadIdx.x;

    sdata[tid] = (i < n) ? g_data[i] : 0.0f;
    __syncthreads();

    // Interleaved addressing: stride grows, active threads shrink
    for (int s = 1; s < blockDim.x; s *= 2) {
        if (tid % (2 * s) == 0) {          // only threads at stride multiples active
            sdata[tid] += sdata[tid + s];  // this causes warp divergence!
        }
        __syncthreads();
    }

    if (tid == 0) g_out[blockIdx.x] = sdata[0];
}

Problem: The condition tid % (2*s) == 0 causes severe warp divergence. At stride s=1: threads 0,2,4,6... are active (50% active). At s=2: threads 0,4,8... (25% active). Also causes shared memory bank conflicts for large strides.

Reduction 2 — Sequential Addressing (No Divergence)

__global__ void reduce2(float* g_data, float* g_out, int n) {
    extern __shared__ float sdata[];
    int tid = threadIdx.x;
    int i   = blockIdx.x * blockDim.x + threadIdx.x;

    sdata[tid] = (i < n) ? g_data[i] : 0.0f;
    __syncthreads();

    // Sequential addressing: stride shrinks from blockDim/2 to 1
    // No divergence within a warp: threads 0..(blockDim/2-1) always active together
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            sdata[tid] += sdata[tid + s];  // no bank conflicts, sequential
        }
        __syncthreads();
    }

    if (tid == 0) g_out[blockIdx.x] = sdata[0];
}

Win: No warp divergence for most steps. No bank conflicts. Remaining problem: When the block launches, half the threads (the upper half) do exactly one load from global memory, then become permanently idle. 50% of thread-cycles wasted.

Reduction 3 — Idle Thread Elimination

__global__ void reduce3(float* g_data, float* g_out, int n) {
    extern __shared__ float sdata[];
    int tid = threadIdx.x;
    // Each thread loads TWO elements and adds them on the way in
    int i   = blockIdx.x * (blockDim.x * 2) + threadIdx.x;

    float val = 0.0f;
    if (i < n)          val += g_data[i];
    if (i + blockDim.x < n) val += g_data[i + blockDim.x];
    sdata[tid] = val;
    __syncthreads();

    // Now do the tree reduction (same as reduce2)
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) sdata[tid] += sdata[tid + s];
        __syncthreads();
    }

    if (tid == 0) g_out[blockIdx.x] = sdata[0];
}

Win: Each thread now does useful work on entry (2 loads + 1 add per thread). The grid size halves (each block handles 2× more elements). Effective memory bandwidth doubles compared to reduce2.

Reduction 4 — Unrolling the Last Warp

// Device function: no sync needed inside a warp (threads execute in lockstep)
__device__ void warpReduce(volatile float* sdata, int tid) {
    sdata[tid] += sdata[tid + 32];
    sdata[tid] += sdata[tid + 16];
    sdata[tid] += sdata[tid +  8];
    sdata[tid] += sdata[tid +  4];
    sdata[tid] += sdata[tid +  2];
    sdata[tid] += sdata[tid +  1];
}

__global__ void reduce4(float* g_data, float* g_out, int n) {
    extern __shared__ float sdata[];
    int tid = threadIdx.x;
    int i   = blockIdx.x * (blockDim.x * 2) + threadIdx.x;

    float val = 0.0f;
    if (i < n)          val += g_data[i];
    if (i + blockDim.x < n) val += g_data[i + blockDim.x];
    sdata[tid] = val;
    __syncthreads();

    for (int s = blockDim.x / 2; s > 32; s >>= 1) {
        if (tid < s) sdata[tid] += sdata[tid + s];
        __syncthreads();
    }

    // Last warp (tid < 32): no syncthreads needed, unroll manually
    if (tid < 32) warpReduce(sdata, tid);

    if (tid == 0) g_out[blockIdx.x] = sdata[0];
}

Win: Eliminates 5 __syncthreads() calls for the last warp. Each sync costs ~20-30 cycles; removing them for the final 6 steps saves up to 150 cycles per block.

Reduction 5 — Complete Unroll (Template)

template <unsigned int blockSize>
__global__ void reduce5(float* g_data, float* g_out, int n) {
    extern __shared__ float sdata[];
    int tid = threadIdx.x;
    int i   = blockIdx.x * (blockSize * 2) + threadIdx.x;

    float val = 0.0f;
    if (i < n)          val += g_data[i];
    if (i + blockSize < n) val += g_data[i + blockSize];
    sdata[tid] = val;
    __syncthreads();

    // Compile-time unrolling: compiler generates separate code for each blockSize
    if (blockSize >= 512) { if (tid < 256) sdata[tid] += sdata[tid+256]; __syncthreads(); }
    if (blockSize >= 256) { if (tid < 128) sdata[tid] += sdata[tid+128]; __syncthreads(); }
    if (blockSize >= 128) { if (tid <  64) sdata[tid] += sdata[tid+ 64]; __syncthreads(); }

    if (tid < 32) {
        if (blockSize >= 64) sdata[tid] += sdata[tid + 32];
        if (blockSize >= 32) sdata[tid] += sdata[tid + 16];
        if (blockSize >= 16) sdata[tid] += sdata[tid +  8];
        if (blockSize >=  8) sdata[tid] += sdata[tid +  4];
        if (blockSize >=  4) sdata[tid] += sdata[tid +  2];
        if (blockSize >=  2) sdata[tid] += sdata[tid +  1];
    }

    if (tid == 0) g_out[blockIdx.x] = sdata[0];
}

// Launch with explicit block size template:
reduce5<256><<<grid, 256, 256*sizeof(float)>>>(d_data, d_out, N);

Reduction 6 — Warp Shuffle (Modern CUDA)

// Warp-level reduction: no shared memory, pure register communication
__device__ float warpReduceSum(float val) {
    for (int offset = 16; offset > 0; offset /= 2)
        val += __shfl_down_sync(0xFFFFFFFF, val, offset);
    return val;  // lane 0 has the sum
}

// Block-level reduction using warp shuffle + minimal shared memory
__global__ void reduce6(float* g_data, float* g_out, int n) {
    __shared__ float warpSums[32];  // one slot per warp (max 32 warps per block)

    int tid = threadIdx.x;
    int i   = blockIdx.x * (blockDim.x * 2) + threadIdx.x;

    float val = 0.0f;
    if (i < n)          val += g_data[i];
    if (i + blockDim.x < n) val += g_data[i + blockDim.x];

    // Phase 1: intra-warp reduction via shuffle (no shared memory needed)
    val = warpReduceSum(val);

    // Phase 2: one value per warp, write to shared memory
    int lane   = tid % 32;
    int warpId = tid / 32;
    if (lane == 0) warpSums[warpId] = val;
    __syncthreads();

    // Phase 3: first warp reduces all warp sums
    int nWarps = blockDim.x / 32;
    val = (tid < nWarps) ? warpSums[tid] : 0.0f;
    if (tid < 32) val = warpReduceSum(val);

    if (tid == 0) g_out[blockIdx.x] = val;
}
Why warp shuffle is better: Instead of writing to shared memory and syncing for the last 6 reduction steps, __shfl_down_sync communicates directly through the warp's register file. No memory bandwidth consumed, no bank conflict possible, no sync needed. It's the fastest intra-warp reduction possible on current NVIDIA hardware.

Two-Pass Reduction for Large Arrays

void reduceArray(float* d_data, float* d_result, int n) {
    int blockSize = 256;
    int gridSize  = (n + blockSize * 2 - 1) / (blockSize * 2);

    float* d_partial;
    cudaMalloc(&d_partial, gridSize * sizeof(float));

    // Pass 1: each block reduces its chunk to one value
    reduce6<<<gridSize, blockSize, 0>>>(d_data, d_partial, n);

    // Pass 2: reduce partial results (usually fits in one block)
    reduce6<<<1, blockSize, 0>>>(d_partial, d_result, gridSize);

    cudaFree(d_partial);
}

CUB: The Production Answer

#include <cub/cub.cuh>

// CUB device-wide reduction (one function call, handles all edge cases)
void cubReduce(float* d_data, float* d_result, int n) {
    void* d_temp = nullptr;
    size_t tempBytes = 0;

    // First call: get required temp storage size
    cub::DeviceReduce::Sum(d_temp, tempBytes, d_data, d_result, n);

    // Allocate temp storage
    cudaMalloc(&d_temp, tempBytes);

    // Second call: actual reduction
    cub::DeviceReduce::Sum(d_temp, tempBytes, d_data, d_result, n);
    cudaDeviceSynchronize();

    cudaFree(d_temp);
}

// CUB also has: Min, Max, ArgMin, ArgMax, custom operators
cub::DeviceReduce::Reduce(d_temp, tempBytes, d_data, d_result, n,
                           cub::Max(), INT_MIN);
